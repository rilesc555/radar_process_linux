RESET:PARAMETERS

## sets SWT operation to track UAVS
MODE:SWT:OPERATIONMODE 1

enable_track
enable_track_logging

## Set orientation in form <heading, pitch, roll>. All in degrees float. 
## Heading not necessary, can be left at zero. 
## Roll should be zero--make sure radar is positioned correctly
## Pitch TBD
PLATFORM:STATE:ORIENTATION eldorado 0, 20, 0

## Set radar height in meters. 
PLATFORM:STATE:ELEVATIONAGL eldorado 2

## Set minimum and maximum height in meters for tracking. Probably best to make this the same as the radar height or slightly above
AGLMASK:MINAGL eldorado 5

AGLMASK:MAXAGL eldorado 500

AGLMASK:ENABLE eldorado TRUE

pause 1000
